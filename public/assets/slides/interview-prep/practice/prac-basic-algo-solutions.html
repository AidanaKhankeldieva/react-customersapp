Find the Highest Value in a given array 
maxValue([1,2,3,4]) // 4


// Solution 1: Loop
function maxValueLoop(arr) {
    let max = arr[0];
    for(let i=1; i < arr.length;i++){
        if (arr[i] > max){
            max = arr[i];
        }
    }
    return max
}
// Time: O(n)

// Solution 2: Sort

function maxValue(arr) {
    arr.sort((a,b) => a-b);
    return arr
}
// Time: O(nlog(n))

// maxValue([4,13,6,2])

Clear all '$' signs from numbers in a given array. => [2,3,4,5,15] 
removePrefix(['$2', '$3', '$4', '$5','$15']) // [2,3,4,5,15]

    
// Solution 1 - map
function removePrefix(arr){
    return arr.map(item=>item.slice(1))
}

    let result = [];
    for(let i=0;i < arr.length;i++){

How would you add your own method to the Array object so the following code would work? hint: prototype 
const arr = [1, 2, 3, 4, 5];
arr.average(); // 3  (15/5) = 3

    
const arr = [1, 2, 3, 4, 5];

Array.prototype.average = function (){
    const sum = this.reduce((a,b)=>a+b);
    return  sum / this.length;
}

arr.average() // (15/5) = 3

Find frequency of 'a' in a given text. 
getFreq('alibaba', 'a') // 3

    
function getFreq(str,char){
    let counter = 0;
    for(let i=0;i < str.length;i++){
        if(str[i] === char){
        counter++
        }
    }
    return counter
    }
    
    getFreq('alibaba', 'a') // 3

Find frequencies of all letters in a given text. 
getFreqAll('alibaba') // {a:3, l:1, i:1, b:2}

    
function getFreqAll(str){
    let result = {}
    for(let i=0;i < str.length;i++){
        if(result.hasOwnProperty(str[i])) {
        result[ str[i] ]++
        } else {
        result[ str[i] ] = 1
        }
    }
    return result 
    }
    
    getFreqAll('alibaba') // {a:3, l:1, i:1, b:2}
